var jsonResponse = pm.response.json();
if (jsonResponse.clientDetails.id === 2){
    pm.environment.set("client_id", jsonResponse.clientDetails.id);
    pm.environment.set("client_secret", jsonResponse.clientDetails.secret);
} else if (jsonResponse.clientDetails.id === 3){
    pm.environment.set("app_client_id", jsonResponse.clientDetails.id);
    pm.environment.set("app_client_secret", jsonResponse.clientDetails.secret);
}
